BalancedGlobalEnergy

Script de Python: Sistema mundial de equilibrio energético desarrollado con tkinter.


import tkinter as tk
from tkinter import ttk
import random
import time
import json
import os
from datetime import datetime

NOMBRE_JUEGO = "BalancedGlobalEnergy"

BASE_DIR = os.path.dirname(os.path.abspath(__file__))
ARCHIVO_PARTIDAS = os.path.join(BASE_DIR, "partidas.json")

CONTINENTES = [
    "América Norte",
    "América Sur",
    "Europa",
    "África",
    "Asia",
    "Oceanía",
    "Antártida"
]

IDEAL = 100 / len(CONTINENTES)

class Juego:
    def __init__(self, root):
        self.root = root
        self.root.title(NOMBRE_JUEGO)
        self.root.geometry("1400x850")
        self.root.configure(bg="#111")
        self.menu_principal()

    def clear(self):
        for widget in self.root.winfo_children():
            widget.destroy()

    def menu_principal(self):
        self.clear()
        frame = tk.Frame(self.root, bg="#111")
        frame.pack(expand=True)

        titulo = tk.Label(
            frame,
            text="BalancedGlobalEnergy",
            font=("Arial", 38, "bold"),
            fg="#00ffcc",
            bg="#111"
        )
        titulo.pack(pady=50)

        subtitulo = tk.Label(
            frame,
            text="Sistema mundial de equilibrio energético",
            font=("Arial", 18),
            fg="white",
            bg="#111"
        )
        subtitulo.pack(pady=10)

        botones = [
            ("INICIAR PARTIDA", self.iniciar_partida),
            ("PARTIDAS GUARDADAS", self.ver_partidas),
            ("SALIR", self.root.quit)
        ]

        for texto, comando in botones:
            tk.Button(
                frame,
                text=texto,
                command=comando,
                font=("Arial", 18, "bold"),
                bg="#00aa88",
                fg="white",
                width=25,
                height=2
            ).pack(pady=15)

    def iniciar_partida(self):
        self.clear()
        self.start_time = time.time()
        self.apagado = random.choice(CONTINENTES)
        self.energias = {}
        restantes = 100
        activos = [c for c in CONTINENTES if c != self.apagado]
        porcentajes = []

        for i in range(len(activos) - 1):
            valor = random.randint(8, 24)
            porcentajes.append(valor)
            restantes -= valor

        porcentajes.append(restantes)
        random.shuffle(porcentajes)

        for i, c in enumerate(activos):
            self.energias[c] = porcentajes[i]

        self.energias[self.apagado] = 0

        notebook = ttk.Notebook(self.root)
        notebook.pack(fill="both", expand=True)

        self.tab_mapa = tk.Frame(notebook, bg="#001219")
        self.tab_pc = tk.Frame(notebook, bg="#111")

        notebook.add(self.tab_mapa, text="Mapa Mundial")
        notebook.add(self.tab_pc, text="Ordenador IA")

        self.crear_mapa()
        self.crear_pc()

    def crear_mapa(self):
        self.timer = tk.Label(
            self.tab_mapa,
            text="Tiempo: 0s",
            font=("Arial", 18, "bold"),
            fg="white",
            bg="#001219"
        )
        self.timer.pack(pady=10)

        self.canvas = tk.Canvas(
            self.tab_mapa,
            width=1300,
            height=700,
            bg="#0a1f2e",
            highlightthickness=0
        )
        self.canvas.pack(pady=10)

        self.actualizar_tiempo()
        colores = {
            "América Norte": "#3498db",
            "América Sur": "#2ecc71",
            "Europa": "#f1c40f",
            "África": "#e67e22",
            "Asia": "#e74c3c",
            "Oceanía": "#9b59b6",
            "Antártida": "#95a5a6"
        }

        def color_cont(cont):
            return "black" if cont == self.apagado else colores[cont]

        self.canvas.create_text(650, 40, text="BALANCED GLOBAL ENERGY MAP", fill="white", font=("Arial", 24, "bold"))

        # Dibujo de Polígonos Simplificados
        self.canvas.create_polygon(80, 120, 240, 80, 310, 150, 290, 260, 180, 320, 100, 250, fill=color_cont("América Norte"), outline="white", width=3)
        self.canvas.create_text(190, 190, text=f"1\nAmérica Norte\n{self.energias['América Norte']}%", fill="white", font=("Arial", 12, "bold"))

        self.canvas.create_polygon(260, 340, 330, 420, 350, 560, 300, 650, 240, 600, 210, 480, fill=color_cont("América Sur"), outline="white", width=3)
        self.canvas.create_text(290, 500, text=f"2\nAmérica Sur\n{self.energias['América Sur']}%", fill="white", font=("Arial", 12, "bold"))

        self.canvas.create_polygon(520, 120, 610, 100, 660, 160, 620, 220, 540, 200, fill=color_cont("Europa"), outline="white", width=3)
        self.canvas.create_text(590, 160, text=f"3\nEuropa\n{self.energias['Europa']}%", fill="white", font=("Arial", 12, "bold"))

        self.canvas.create_polygon(560, 260, 670, 280, 720, 420, 640, 560, 540, 520, 500, 360, fill=color_cont("África"), outline="white", width=3)
        self.canvas.create_text(620, 400, text=f"4\nÁfrica\n{self.energias['África']}%", fill="white", font=("Arial", 12, "bold"))

        self.canvas.create_polygon(760, 120, 1050, 100, 1180, 240, 1110, 380, 980, 400, 850, 330, 720, 220, fill=color_cont("Asia"), outline="white", width=3)
        self.canvas.create_text(950, 250, text=f"5\nAsia\n{self.energias['Asia']}%", fill="white", font=("Arial", 12, "bold"))

        self.canvas.create_polygon(1030, 470, 1130, 500, 1160, 580, 1080, 620, 980, 570, fill=color_cont("Oceanía"), outline="white", width=3)
        self.canvas.create_text(1080, 560, text=f"6\nOceanía\n{self.energias['Oceanía']}%", fill="white", font=("Arial", 12, "bold"))

        self.canvas.create_rectangle(350, 650, 950, 690, fill=color_cont("Antártida"), outline="white", width=3)
        self.canvas.create_text(650, 670, text=f"7 Antártida {self.energias['Antártida']}%", fill="white", font=("Arial", 12, "bold"))

    def crear_pc(self):
        titulo = tk.Label(self.tab_pc, text="SISTEMA DE IA ENERGÉTICA", font=("Consolas", 28, "bold"), fg="#00ff99", bg="#111")
        titulo.pack(pady=20)
        info = tk.Label(self.tab_pc, text=f"CONTINENTE APAGADO: {self.apagado}", font=("Arial", 18, "bold"), fg="white", bg="#111")
        info.pack(pady=10)

        pantalla = tk.Frame(self.tab_pc, bg="black", bd=5, relief="ridge")
        pantalla.pack(pady=20)

        texto = tk.Text(pantalla, width=60, height=15, bg="black", fg="#00ff99", font=("Consolas", 12))
        texto.pack()
        texto.insert(tk.END, "=== RED ELÉCTRICA GLOBAL ===\n\n")

        for i, cont in enumerate(CONTINENTES):
            texto.insert(tk.END, f"{i+1}. {cont} --> {self.energias[cont]}%\n")

        texto.insert(tk.END, f"\nOBJETIVO IDEAL: {round(IDEAL, 2)}% POR CONTINENTE")
        texto.config(state="disabled")

        tk.Label(self.tab_pc, text="Número del continente origen", font=("Arial", 14), fg="white", bg="#111").pack()
        self.input_cont = tk.Entry(self.tab_pc, font=("Arial", 16))
        self.input_cont.pack(pady=10)

        tk.Label(self.tab_pc, text="Porcentaje de energía a transferir", font=("Arial", 14), fg="white", bg="#111").pack()
        self.input_energia = tk.Entry(self.tab_pc, font=("Arial", 16))
        self.input_energia.pack(pady=10)

        tk.Button(self.tab_pc, text="EJECUTAR IA", font=("Arial", 18, "bold"), bg="#00aa88", fg="white", padx=30, pady=15, command=self.resolver).pack(pady=30)

    def actualizar_tiempo(self):
        try:
            tiempo = int(time.time() - self.start_time)
            if self.timer.winfo_exists():
                self.timer.config(text=f"Tiempo: {tiempo}s")
                self.root.after(1000, self.actualizar_tiempo)
        except: pass

    def resolver(self):
        try:
            indice = int(self.input_cont.get()) - 1
            cantidad = float(self.input_energia.get())
            if indice < 0 or indice >= len(CONTINENTES): return
            origen = CONTINENTES[indice]
            if origen == self.apagado or cantidad <= 0 or self.energias[origen] < cantidad: return

            self.energias[origen] -= cantidad
            self.energias[self.apagado] += cantidad

            diferencia = sum(abs(valor - IDEAL) for valor in self.energias.values())
            precision = max(0, int(100 - diferencia * 2))
            tiempo = int(time.time() - self.start_time)

            self.guardar_partida(tiempo, precision)
            self.final(tiempo, precision)
        except Exception as e: print("ERROR:", e)

    def guardar_partida(self, tiempo, precision):
        datos = []
        try:
            if os.path.exists(ARCHIVO_PARTIDAS):
                with open(ARCHIVO_PARTIDAS, "r", encoding="utf-8") as f:
                    datos = json.load(f)
        except: pass

        ahora = datetime.now()
        datos.append({
            "fecha": ahora.strftime("%Y-%m-%d"),
            "hora": ahora.strftime("%H:%M:%S"),
            "tiempo": tiempo,
            "precision": precision
        })

        try:
            with open(ARCHIVO_PARTIDAS, "w", encoding="utf-8") as f:
                json.dump(datos, f, indent=4, ensure_ascii=False)
        except: pass

    def final(self, tiempo, precision):
        self.clear()
        frame = tk.Frame(self.root, bg="#111")
        frame.pack(fill="both", expand=True)
        tk.Label(frame, text="MISIÓN COMPLETADA", font=("Arial", 34, "bold"), fg="#00ff99", bg="#111").pack(pady=50)
        tk.Label(frame, text=f"Tiempo empleado: {tiempo} segundos", font=("Arial", 20), fg="white", bg="#111").pack(pady=10)
        tk.Label(frame, text=f"Precisión energética: {precision}%", font=("Arial", 20), fg="white", bg="#111").pack(pady=10)
        tk.Button(frame, text="VOLVER AL MENÚ", font=("Arial", 16, "bold"), bg="#00aa88", fg="white", command=self.menu_principal).pack(pady=40)

    def ver_partidas(self):
        self.clear()
        frame = tk.Frame(self.root, bg="#111")
        frame.pack(fill="both", expand=True)
        tk.Label(frame, text="PARTIDAS GUARDADAS", font=("Arial", 28, "bold"), fg="white", bg="#111").pack(pady=20)
        
        lista = tk.Listbox(frame, width=90, height=20, font=("Consolas", 12))
        lista.pack(pady=20)

        try:
            with open(ARCHIVO_PARTIDAS, "r", encoding="utf-8") as f:
                datos = json.load(f)
            for p in datos:
                lista.insert(tk.END, f"{p['fecha']} {p['hora']} | Tiempo: {p['tiempo']}s | Precisión: {p['precision']}%")
        except:
            lista.insert(tk.END, "No hay partidas registradas.")

        tk.Button(frame, text="VOLVER", font=("Arial", 16, "bold"), bg="#00aa88", fg="white", command=self.menu_principal).pack(pady=20)

if __name__ == "__main__":
    root = tk.Tk()
    Juego(root)
    root.mainloop()
